% ============================================================
% HANGING QUOTATION MARKS (Solution B: expl3 hangcentquote)
% ============================================================
% This environment handles multi-line quotes with hanging punctuation.
% The opening quote hangs into the left margin, closing quote hangs right.
% Usage: \begin{hangquote}{lang} ... \end{hangquote}
% where lang is: en, de, he, fr, gr

\ExplSyntaxOn

\dim_new:N \l_hcq_gap_dim
\dim_set:Nn \l_hcq_gap_dim { .08em }

\NewDocumentCommand \HangQuoteGap { m }
  { \dim_set:Nn \l_hcq_gap_dim {#1} }

\NewDocumentEnvironment {hangcentquote} { m m +b }
  {
    \seq_set_split:Nnn \l_tmpa_seq { \\ } { #3 }
    \int_compare:nNnTF { \seq_count:N \l_tmpa_seq } = {1}
      {
        \tl_set:Nx \l_tmpa_tl { \seq_item:Nn \l_tmpa_seq {1} }
        \begin{tabular}{@{}r@{}}
          \llap{#1\kern\l_hcq_gap_dim}\l_tmpa_tl\rlap{\kern\l_hcq_gap_dim#2}
        \end{tabular}
      }
      {
        \seq_pop_left:NN  \l_tmpa_seq \l_tmpa_tl
        \seq_pop_right:NN \l_tmpa_seq \l_tmpb_tl
        \begin{tabular}{@{}r@{}}
          \llap{#1\kern\l_hcq_gap_dim}\l_tmpa_tl\\
          \seq_map_inline:Nn \l_tmpa_seq { ##1\\ }
          \l_tmpb_tl\rlap{\kern\l_hcq_gap_dim#2}
        \end{tabular}
      }
  }
  { }

% ============================================================
% HANGING QUOTATION MARKS (Solution E: hangleftquote)
% ============================================================
% Left-aligned text with opening quote hanging LEFT (classical).
% Closing quote sits inline at end of last line.
% Block is meant to be placed inside \begin{flushright}...\end{flushright}
% so the whole block sits flush-right on the page.
% Usage: \begin{hangleftquote}{OpenMark}{CloseMark} ... \end{hangleftquote}

\NewDocumentEnvironment {hangleftquote} { m m +b }
  {
    \seq_set_split:Nnn \l_tmpa_seq { \\ } { #3 }
    \int_compare:nNnTF { \seq_count:N \l_tmpa_seq } = {1}
      {
        \tl_set:Nx \l_tmpa_tl { \seq_item:Nn \l_tmpa_seq {1} }
        \begin{tabular}{@{}l@{}}
          \llap{#1\kern\l_hcq_gap_dim}\l_tmpa_tl #2
        \end{tabular}
      }
      {
        \seq_pop_left:NN  \l_tmpa_seq \l_tmpa_tl
        \seq_pop_right:NN \l_tmpa_seq \l_tmpb_tl
        \begin{tabular}{@{}l@{}}
          \llap{#1\kern\l_hcq_gap_dim}\l_tmpa_tl\\
          \seq_map_inline:Nn \l_tmpa_seq { ##1\\ }
          \l_tmpb_tl #2
        \end{tabular}
      }
  }
  { }

    \end{minipage}
  \end{flushright}
}

% Topic map macro: render as small caps with ○ separator
\newcommand{\topicmap}[1]{%
  \noindent\textsc{%
    \def\tempa{}%
    \foreach \x in {#1} {%
      \ifx\tempa\empty
        \x
      \else
        \enspace$\circ$\enspace\x
      \fi
      \xdef\tempa{nonempty}%
    }%
  }%
}

\newcommand{\inputstory}[1]{%
    % === CHAPTER STRUCTURE: EXACTLY 10 PAGES ===
    % 1. Big Title (recto) - 1 page
    % 2. Sidenote (verso) - 1 page  
    % 3. Title+Summary+Topicmap+Quote (recto) - 1 page
    % 4-8. Historical+Main content - 5 pages
    % 9. Technical - 1 page
    % 10. Empty verso page - 1 page (for chapter separation)
    %
    % VERSO EMPTY PAGE BEFORE EACH CHAPTER (except first - it follows mainmatter)
    % This ensures proper recto/verso alignment and 10-page structure
    \iffirstchapter
        % First chapter: just start on new page, no blank separator
        \clearpage
        \global\firstchapterfalse
    \else
        % Subsequent chapters: add blank verso separator page
        \clearpage
        \thispagestyle{empty}
        \mbox{}
        \clearpage
    \fi

    % --- PAGE 1: Dedicated Title Page (big, centered) ---
    \phantomsection
    % Place chapter label and TOC entry HERE (page 1) so page numbers are correct
    \expandafter\ifx\expandafter\\\storedchapterlabel\\\else\label{\storedchapterlabel}\fi%
    \addcontentsline{toc}{chapter}{%
      \protect\numberline{\thechapter}\textcolor{toctitlecolor}{\storedchaptertitle}\\
      {\normalfont\small\textit{\textcolor{summarycolor}{\storedchaptersummary}}}%
    }%
    \readfirstline{\chaptertitle}{#1/title.tex}
    \readfirstline{\chaptersummary}{#1/summary.tex}

    \thispagestyle{empty}
    \begin{center}
        \vspace*{\fill}
        {\fontsize{48pt}{62pt}\selectfont\bfseries\raggedright
        \parbox{0.8\textwidth}{\centering\input{#1/title}}}
        \vspace*{\fill}
    \end{center}

    \clearpage

    % --- PAGE 2: Sidenote ---
    \IfFileExists{#1/sidenote.tex}{%
        \input{#1/sidenote}%
    }{%
        % If no sidenote, create empty page
        \thispagestyle{empty}
        \mbox{}
    }{}

    \clearpage

    % --- PAGE 3: Title + Summary + Topicmap + Quote (original intro style) ---
    \thispagestyle{empty}
    \begin{center}
        \vspace*{0.5em}

        {\Huge \bfseries \input{#1/title}}

        \vspace{1.5em}
        \begin{minipage}{0.82\textwidth}
            {\fontsize{13pt}{18pt}\selectfont\color{black}
            \sloppy
            \justifying
            \input{#1/summary}}
        \end{minipage}

        \vspace{1.5em}

        \chapterseparator

        \vspace{1.5em}

        \IfFileExists{#1/topicmap.tex}{%
            \begin{minipage}{0.7\textwidth}
                \centering
                 \input{#1/topicmap}
            \end{minipage}
        }{}

        \vfill

        \IfFileExists{#1/quote.tex}{%
            \vspace{1em}
            \begin{minipage}{0.85\textwidth}
                \centering \input{#1/quote}
            \end{minipage}
        }{}

        \vspace*{0.5em}
    \end{center}

    \clearpage

    % --- PAGES 4-8: Historical + Main + Optional materials (exactly 5 pages total) ---
    % Set chapter header HERE when content actually begins
    \chaptermark{\storedchaptertitle}
    \chapterdoi{\thechapter}
    {\LARGE \bfseries \input{#1/title}}
    \input{#1/historical}
    \input{#1/main}
    \IfFileExists{#1/phenomenon_extra.tex}{\input{#1/phenomenon_extra}}{}
    
    % Include optional materials as part of the 5-page content block
    \IfFileExists{#1/joke.tex}{%
        \input{#1/joke}%
    }{}
    
    \IfFileExists{#1/exercises.tex}{%
        \input{#1/exercises}%
    }{}
    
    \IfFileExists{#1/cartoon.tex}{%
        \input{#1/cartoon}%
    }{}
    
    \IfFileExists{#1/imagefigure.tex}{%
        \input{#1/imagefigure}%
    }{}

    % --- PAGE 9: Technical (exactly 1 page) ---
    \newpage
    \input{#1/technical}

    % --- NO PAGE 10 EMPTY PAGE HERE ---
    % Empty verso pages are now added BEFORE each chapter (see top of \inputstory)
    % This prevents cumulative page drift and maintains 10-page alignment
    % Last chapter will not have trailing empty page


}
